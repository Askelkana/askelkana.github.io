/**
 * @namespace Object
 * @desc Additional utility methods.
 */
;
(function () {
	Object.defineProperties(Object.prototype, {
		/**
		 * Iterates over the properties of the object, calling the passed callback for each property. Behaves in
		 * similar fashion to the <code>Array.prototype.forEach</code> method.
		 *
		 * @param {function} callback
		 * The iteration callback. Called as follows:
		 * <pre>
		 *     callback(key, value)
		 * </pre>
		 * And where <code>this</code> is the target object.
		 *
		 * @param {boolean} [includePrototype=false]
		 * If true, then properties included in the object's prototype chain are also processed.
		 *
		 * @returns {undefined}
		 * @function Object#forEach
		 */
		forEach: {
			value: function (callback, includePrototype) {
				var self = this;
				if (is.not.function(callback)) {
					throw new TypeError('function expected.');
				}
				if (!!includePrototype) {
					for (var key in self) {
						processKey(key);
					}
				} else {
					Object.keys(self).forEach(processKey);
				}

				function processKey(key) {
					callback.call(self, key, self[key]);
				}
			}
		}
	});
}());
