/**
 * @namespace Number
 * @desc Additional utility methods.
 */
;
(function () {
	"use strict";

	Object.defineProperties(Number.prototype, {
		/**
		 * @function Number#inside
		 * @desc Determines if the number is within the given bounds.
		 *
		 * @param {number} [min=-Infinity]
		 * The lower bound.
		 *
		 * @param {number} [max=Infinity]
		 * The upper bound.
		 *
		 * @returns {boolean}
		 * @throws TypeError
		 */
		inside:   {
			value: function (min, max) {
				min = min || -Infinity;
				max = max || Infinity;
				if (is.not.number(min) || is.not.number(max)) {
					throw new TypeError('number expected.');
				}
				return this.valueOf() > Math.min(min, max) && this.valueOf() < Math.max(min, max);
			}
		},
		/**
		 * @function Number#integer
		 * @desc Determines if the number is a whole number.
		 *
		 * @returns {boolean}
		 */
		integer:  {
			value: function () {
				return this == Math.floor(this);
			}
		},
		/**
		 * @function Number#negative
		 * @desc Determines if the number is a less than zero.
		 *
		 * @returns {boolean}
		 */
		negative: {
			value: function () {
				return this < 0;
			}
		},
		/**
		 * @function Number#outside
		 * @desc Determines if the number is outside the given bounds.
		 *
		 * @param {number} [min=-Infinity]
		 * The lower bound.
		 *
		 * @param {number} [max=Infinity]
		 * The upper bound.
		 *
		 * @returns {boolean}
		 * @throws TypeError
		 */
		outside:  {
			value: function (min, max) {
				min = min || -Infinity;
				max = max || Infinity;
				if (is.not.number(min) || is.not.number(max)) {
					throw new TypeError('number expected.');
				}
				return this.valueOf() < Math.min(min, max) || this.valueOf() > Math.max(min, max);
			}
		},
		/**
		 * @function Number#positive
		 * @desc Determines if the number is greater than zero.
		 *
		 * @returns {boolean}
		 */
		positive: {
			value: function () {
				return this > 0;
			}
		},
		/**
		 * @function Number#rounded
		 * @desc Rounds the number to the nearest integer.
		 *
		 * @returns {number}
		 */
		rounded:  {
			value: function () {
				return Math.round(this);
			}
		},
		/**
		 * @function Number#zero
		 * @desc Determines if the number is zero.
		 *
		 * @returns {boolean}
		 */
		zero:     {
			value: function () {
				return this.valueOf() === 0;
			}
		}
	});
}());
