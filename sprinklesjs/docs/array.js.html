/**
 * @namespace Array
 * @desc Additional utility methods.
 */
;
(function () {
	Object.defineProperties(Array.prototype, {
		/**
		 * Determines if the passed value is found in the calling array.
		 *
		 * @param {*} x
		 * The value to look for.
		 *
		 * @returns {boolean}
		 * @function Array#contains
		 */
		contains:  {
			value: function (x) {
				return this.indexOf(x) !== -1;
			}
		},
		/**
		 * Determines if the passed value is not found in the calling array.
		 *
		 * @param {*} x
		 * The value to look for.
		 *
		 * @returns {boolean}
		 * @function Array#excludes
		 */
		excludes: {
			value: function (x) {
				return this.indexOf(x) === -1;
			}
		},
		/**
		 * Calculates the intersect of the calling and passed arrays, based solely on the presence within both arrays of
		 * common values.
		 *
		 * @param {array} B
		 * The array to intersect with.
		 *
		 * @returns {array} The values contained in both arrays.
		 * @function Array#intersect
		 */
		intersect: {
			value: function (b) {
				var a = this;
				var rv = [];
				if (is.not.array(b)) {
					throw new TypeError('array expected.');
				}
				a.forEach(function (v, i) {
					if (b.contains(v)) {
						rv.push(v);
					}
				});
				return rv;
			}
		},
		/**
		 * Returns an array with all duplicate elements from the calling array removed.
		 *
		 * @returns {array} The values of the array, occuring only once.
		 * @function Array#unique
		 */
		unique:    {
			value: function () {
				var rv = [];
				this.forEach(function (v, i) {
					if (rv.excludes(v)) {
						rv.push(v);
					}
				});
				return rv;
			}
		}
	});
}());
